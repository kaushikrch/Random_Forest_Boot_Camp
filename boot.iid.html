## This is a naive version of bootstrapping but may be useful for novices.
library(animation)
library(slidifyLibraries)
library(ggplot2)
library(MASS)
library(tree)
library(ISLR)
library(slidify)
par(mar = c(2.5, 4, 0.5, 0.5))
ani.options(nmax = ifelse(interactive(), 50, 10))
boot.iid(x = rnorm(25, 0, 1), statistic = mean, m = 25, main = c("", 
    ""), heights = c(1, 2))
